claude-teams-operator {{ .Chart.AppVersion }} installed in namespace {{ .Release.Namespace }}.

# 1. Verify the operator is running
kubectl rollout status deployment/{{ .Release.Name }}-controller-manager \
  --namespace {{ .Release.Namespace }} --timeout=120s

# 2. Confirm the CRDs are installed
kubectl get crds | grep claude.amcheste.io

# 3. Create your Anthropic API key secret (replace sk-ant-... with your key)
kubectl create secret generic anthropic-api-key \
  --from-literal=ANTHROPIC_API_KEY=sk-ant-... \
  --namespace <your-team-namespace>

# 4. Apply your first AgentTeam
kubectl apply -f - <<'YAML'
apiVersion: claude.amcheste.io/v1alpha1
kind: AgentTeam
metadata:
  name: hello-team
spec:
  auth:
    apiKeySecret: anthropic-api-key
  lead:
    model: opus
    prompt: "Coordinate the worker."
  teammates:
    - name: worker
      model: sonnet
      prompt: "Say hello."
YAML

# 5. Watch progress
kubectl get agentteam hello-team -w
kubectl describe agentteam hello-team
